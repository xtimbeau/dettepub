data_graph <- tsh |> 
  pluck("data") |> 
  filter(date >= "2010-01-01") |> 
  mutate( date  = if_else(date<= now() - months(1), floor_date(date, unit = "month") + days(15), date)) |>
  group_by(date, pays) |> 
  summarize(taux = mean(taux, na.rm = TRUE),
         mois = n()>1,
         .groups = "drop") |>
  mutate(
    geo = case_match(pays,
                     "france" ~ "FR",
                     "germany" ~ "DE",
                     "italy" ~ "IT"),
    tooltip = str_c("<b>", pays, "</b><br>",
                    if_else(mois, date_mois(date), date_jour(date)),
                    " : ", round(taux,2), "%"))

gspread <-  ggplot(data_graph) +
  aes(x=date, y=taux, color=geo, group=geo) +
  geom_line(linewidth = 0.5, show.legend = FALSE) +
  geom_point_interactive(aes(tooltip = tooltip, data_id = date, fill = geo),
                         stroke = 0.5, col = "white", size = 1, shape=21,
                         hover_nearest = TRUE) +
  labs(y="Taux, spread (en %)",
    x = NULL,
    fill=NULL)+
  ofce::scale_color_pays("eurostat") +
  ofce::scale_ofce_date(
    date_breaks = "2 year", 
    date_minor_breaks = "1 month",
    limits = c(ymd("2010-01-01"), NA)) +
   guides(fill = marquee::guide_marquee(
     "!!1, !!2, !!3") ) +
  ofce_caption(
    source="investing.com",
    note = "données téléchargées le {date_jour(tsh$date)}",
    dpt = data_graph$date, dptf = "day")

girafy(gspread)
